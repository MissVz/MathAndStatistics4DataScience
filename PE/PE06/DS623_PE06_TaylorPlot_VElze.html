def taylor_series(f, x, n):
    taylor_approx = np.zeros_like(x)
    for k in range(n + 1):
        if k % 4 == 0:
            coeff = 1
        elif k % 4 == 1:
            coeff = -1 if k % 2 == 1 else 1
        elif k % 4 == 2:
            coeff = -1
        else:
            coeff = 1 if k % 2 == 1 else -1

        taylor_approx += coeff * x**k / factorial(k)
    return taylor_approx
